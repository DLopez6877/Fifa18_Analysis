def extract_value_from(value):
    out = value.replace('€', '')
    if 'M' in out:
        out = float(out.replace('M', ''))*1000000
    elif 'K' in value:
        out = float(out.replace('K', ''))*1000
    return float(out)

Using the list of high accuracy players, create a function that will seperate players preferred positions into catagories and add them to a dictionary

In [17]:
positions = {}

def sort_positions(x):
    split_pos = x.split(' ')
    for pos in split_pos:
        if (pos == ''):
            continue
        else:
            if (pos not in positions):
                positions[pos] = 1
            else:
                positions[pos] += 1;  

df3['Preferred Positions'].apply(lambda x: sort_positions(x))
